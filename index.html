📥 Importar Dados do BigQuery

Execute a query abaixo no BigQuery, copie o resultado e cole na área de texto.
-- Execute no BigQuery e cole o resultado abaixo WITH BASE AS ( -- Cole aqui a CTE BASE_NPS_Y20_DETAIL da sua query original ... ) SELECT NP.USER_TEAM_NAME, IF(NP.USER_TEAM_ID = 2352, 'C2C', NP.USER_TEAM_CHANNEL) AS USER_TEAM_CHANNEL, NP.USER_OFFICE, FORMAT_DATE('%Y-%m-%d', DATE_TRUNC(NP.RES_END_DATE, MONTH)) AS PERIODO, 'MES' AS TIPO, COUNT(*) AS ENCUESTAS, ROUND(AVG(NP.GAP_TGT), 2) AS GAP_TGT, ROUND((SUM(NP.PROMOTER) - SUM(NP.DETRACTOR)) * 100.0 / COUNT(*), 2) AS NPS, ROUND(AVG(NP.SURVEY_TARGET_VALUE), 2) AS TARGET FROM BASE NP GROUP BY 1,2,3,4,5 UNION ALL SELECT NP.USER_TEAM_NAME, IF(NP.USER_TEAM_ID = 2352, 'C2C', NP.USER_TEAM_CHANNEL), NP.USER_OFFICE, FORMAT_DATE('%Y-%m-%d', DATE_TRUNC(NP.RES_END_DATE, ISOWEEK)) AS PERIODO, 'SEMANA' AS TIPO, COUNT(*), ROUND(AVG(NP.GAP_TGT), 2), ROUND((SUM(NP.PROMOTER) - SUM(NP.DETRACTOR)) * 100.0 / COUNT(*), 2), ROUND(AVG(NP.SURVEY_TARGET_VALUE), 2) FROM BASE NP WHERE DATE_TRUNC(NP.RES_END_DATE, ISOWEEK) >= DATE_SUB(CURRENT_DATE(), INTERVAL 3 WEEK) GROUP BY 1,2,3,4,5 ORDER BY 1,2,3,4
